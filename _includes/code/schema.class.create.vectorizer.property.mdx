let classObj = {
    'class': 'Article',
    'properties': [
        {
            'dataType': ['text'],
            'name': 'title',
            // highlight-start
            'moduleConfig': {
                'text2vec-openai': {
                    'skip': False,
                    'vectorizePropertyName': False
                }
            }
            // highlight-end
        },
    ],
    'vectorizer': 'text2vec-openai'
}

// add the schema
client
  .schema
  .classCreator()
  .withClass(classObj)
  .do()
  .then(res => {
    console.log(res)
  })
  .catch(err => {
    console.error(err)
  });
